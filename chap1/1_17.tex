\problem{Let $g:\mathbb{N}\to\mathbb{N}\in\PRF,
f:\mathbb{N}^2\to\mathbb{N}$ satisfies that
$f(x,0) = g(x)$, $f(x, y + 1) = f(f(\ldots f(f(x, y), y - 1), \ldots),0)$.
Prove that $f\in\PRF$.}

\begin{pf} \rm
 Let $G(x, 0) = x$ and $G(x, y + 1) = g(G(x, y))$.
 A simple induction shows that $f(x, y) = g^{2^{y-1}}(x)$,
 thust $f(x,y) = G(x, 2^{y-1}) \in\PRF$. \qed
\end{pf}
