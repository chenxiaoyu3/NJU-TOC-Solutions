\problem{Let $p$ be a polynomial with integral coefficient, and $f:\mathbb{N}\to\mathbb{N}$
defined by the non-negative root of $f(a) = p(x) - a$. Prove that $f\in\RF$.}

\begin{pf} \rm
 Let $p(x) = a_n x^n + \ldots + a_1 x + a_0$, 
 $S=\{i~|~a_i > 0\}, $ and $T =\{i~|~a_i < 0\}$, we have

\[\displaystyle |p(x) - a| = 
 \left|
  \sum_{i\in S} |a_i| x^i - \left( a + \sum_{i\in T} |a_i| x^i \right)
 \right| \in \EF.\]

\noindent Therefore, $f(a) = \mu x. |p(x) - a|\in\RF$. \qed
\end{pf}
