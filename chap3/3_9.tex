\problem{Prove that for all $P,Q\in\Lambda$, $P\equivbeta Q$ implies the existence of
$n\geq 0$ and $P_0,\ldots,P_n\in\Lambda$ satisfying $P\equiv P_0$,
$Q\equiv P_n$ and $P_i\onestepbeta P_{i+1}$ 
or $P_{i+1}\onestepbeta P_i$ for all $i<n$.}
\begin{pf} \rm The technique used is exactly the same as in problem 3.7,
except we are using
$\displaystyle \equivbeta =
 \bigcup_{i=0}^{\infty} \left(\onestepbeta \cup \onestepbeta^{-1} \right)^i
$. \qed
\end{pf}
